1. A rendszer célja
Maga a weboldal a Star Wars világáról fog információkat tartalmazni. Listaként megfogja jeleníteni a különböző élőlényeket, bolygókat, űrhajókat és fegyvereket, valamint az eddig megjelent játékokat. Ezeket az emberek fogják tudni módosítani, törölni az adatokat, kilistázni őket, valamint újabb adatokkal bővíteni a listákat. Plusz információkat fog még tartalmazni a filmekről és sorozatokról, amiket nem lehet módosítani.

5. Funkcionális terv
A főoldalon legyen egy menü, amely segítségével a felhasználók tudnak navigálni a lapok között. Az élőlényeknek, űrhajóknak, bolygóknak és a fegyvereknek, valamint a játékoknak legyenek külön-külön táblái az adatbázisban. Ezeket a felhasználók tudják szerkeszteni, listázni, törölni, valamint új még meg nem lévő adatokat hozzáadni. Az eddig megjelent filmekről és sorozatokról is legyenek külön-külön lapok, melyeken információk legyenek az adott filmről vagy sorozatról kapcsolódóan. Ezeket a felhasználok ne tudják szerkeszteni.
6. Fizikai környezet
NetBeans-ben a CodeIgniter keretrendszert felhasználva lesz elkészítve a weboldal. A felhasznált nyelvek: PHP, HTML és CSS.
7. Adatbázis terv
CREATE TABLE IF NOT EXISTS jedi(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    homeworld VARCHAR(250) NOT NULL,
    species VARCHAR(250) NOT NULL,
    gender VARCHAR(250) NOT NULL,
    rank VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_jedi PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS sith(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    homeworld VARCHAR(250) NOT NULL,
    species VARCHAR(250) NOT NULL,
    gender VARCHAR(250) NOT NULL,
    rank VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_sith PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS droid(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    homeworld VARCHAR(250) NOT NULL,
    creator VARCHAR(250) NOT NULL,
    model VARCHAR(250) NOT NULL,
    class VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_droid PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS planets(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    regio VARCHAR(250) NOT NULL,
    sector VARCHAR(250) NOT NULL,
    system VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_planets PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS weapons(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    type VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_weapons PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS vehicles(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    model VARCHAR(250) NOT NULL,
    class VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_vehicles PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS species(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_species PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS senators(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    homeworld VARCHAR(250) NOT NULL,
    species VARCHAR(250) NOT NULL,
    gender VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,
    file VARCHAR(250) NOT NULL,

    CONSTRAINT PK_senators PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS games(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(250) NOT NULL,
    developres VARCHAR(250) NOT NULL,
    publishers VARCHAR(250) NOT NULL,
    release_date DATE NOT NULL,
    models VARCHAR(250) NOT NULL,
    platforms VARCHAR(250) NOT NULL,
    image VARCHAR(250) NOT NULL,

    CONSTRAINT PK_games PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS films(
    id INT NOT NULL AUTO_INCREMENT,
    title VARCHAR(250) NOT NULL,
    release_date DATE NOT NULL,
    image VARCHAR(250) NOT NULL,

    CONSTRAINT PK_films PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS series(
    id INT NOT NULL AUTO_INCREMENT,
    title VARCHAR(250) NOT NULL,
    release_date DATE NOT NULL,
    image VARCHAR(250) NOT NULL,

    CONSTRAINT PK_series PRIMARY KEY(id)
);
